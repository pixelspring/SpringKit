            </tr>
            <tr>
                <td>$card-border</td>
                <td>Boolean</td>
                <td><code>True</code></td>

            </tr>
            <tr>
                <td>$card-background</td>
                <td>Color</td>
                <td><code>$ghost</code></td>

            </tr>
            <tr>
                <td>$card-padding</td>
                <td>Number</td>
                <td><code>1rem</code></td>
                <td>Defines the padding of the card elements</td>

            <tr>
                <td>$card-header-bg</td>

<h3>Example Code:</h3>
{% highlight html %}
<!-- Set element to primary color bg -->
.element-primary {
    background: palette(primary);
}

<!-- Set element to primary light color bg -->
.element-primary-light {
    background: palette(primary, light);
}
{% endhighlight %}
